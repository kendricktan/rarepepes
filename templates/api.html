import base64
import requests
import json

from io import BytesIO
from PIL import Image

# YOUR IMAGE FILE TO BE PEPE'ED
img_to_be_posted = open('IMAGE.png', 'rb').read()

# POST request to the server
r = requests.post('https://makerarepepes.me/generate',
	# Payload
	data = {
		'img': base64.b64encode(img_to_be_posted).decode('utf-8')
	}
)


# Read base64 encoded image from the server
img_base64 = json.loads(r.content)['img']

# Construct a Pillow Image Class
img = Image.open(BytesIO(base64.b64decode(img_base64)))

# Show image
img.show()
